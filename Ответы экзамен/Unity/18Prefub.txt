public GameObject prefab;
    void Start()
    {
    }

    void Update()
    {
        if (Input.GetKey(KeyCode.U))
        {
            Vector3 position = new Vector3(Random.Range(-5f, 5f), Random.Range(1f, 10f), Random.Range(1f, 10f));
            Instantiate(prefab, position, Quaternion.identity);
        }
    }
