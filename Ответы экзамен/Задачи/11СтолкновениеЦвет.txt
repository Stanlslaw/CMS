using System;using System.Collections;using System.Collections.Generic;using System.Linq;using UnityEngine;using UnityEngine.UI;   public class SCRIPT : MonoBehaviour{    private Renderer objectRenderer;    private Color originalColor;    void Start()    {        objectRenderer = GetComponent<Renderer>();                    originalColor = objectRenderer.material.color;    }    void Update()    {        if (Input.GetKey(KeyCode.D))        {            transform.Translate(Vector3.back * 1f * Time.deltaTime);        }    }    void OnCollisionEnter(Collision collision)    {        if (collision.gameObject.CompareTag("Obstacle"))        {            ChangeColor();        }    }    void OnCollisionExit(Collision collision)    {        if (collision.gameObject.CompareTag("Obstacle"))        {            objectRenderer.material.color = originalColor;        }    }    void ChangeColor()    {        objectRenderer.material.color = UnityEngine.Random.ColorHSV();    }}
